CREATE TABLE IF NOT EXISTS user(user_id INTEGER PRIMARY KEY AUTOINCREMENT, first_name TEXT, last_name TEXT, username TEXT NOT NULL, password TEXT, UNIQUE(user_id, username));
CREATE TABLE IF NOT EXISTS habit(habit_id INTEGER PRIMARY KEY AUTOINCREMENT, user_id INTEGER, habit_name TEXT, habit_description TEXT, creation_date DATE DEFAULT CURRENT_DATE , periodicity TEXT, active INTEGER, type TEXT, FOREIGN KEY (user_id) REFERENCES user (user_id), UNIQUE(habit_id, habit_name, habit_description));
CREATE TABLE IF NOT EXISTS predefined_habit(habit_name TEXT, habit_description TEXT, creation_date DATE DEFAULT CURRENT_DATE , periodicity TEXT, active INTEGER, type TEXT);
CREATE TABLE IF NOT EXISTS progress(habit_id INTEGER, progress_date DATE DEFAULT CURRENT_DATE, status INTEGER, FOREIGN KEY (habit_id) REFERENCES habit (habit_id));
